Project online_store {
  database_type: 'multi database'
  Note: 'Schema of each database'
}

Table accounts {
  "user id" integer [primary key]
  name string
  
    indexes {
    "user id" [pk]
  }
}

Table carts {
  "cart id" integer [primary key]
  "product id" string [primary key]
  quantity integer
}

Table "orders metadata" {
  "user id" integer [primary key]
  "purchase date" timestamp [primary key]
  "order id"  integer [primary key]
}

Table orders {
  "order id"  integer [primary key]
  "product id" string [primary key]
  "quantity" string
}

Table products {
  "product id" integer [primary key]
  title string
  description string
  rating integer
  
}

Table "product tag mapping" {
  "product id" integer [primary key]
  "tag id" integer [primary key]
  
}

Table "product tags" {
  "tag id" integer [primary key]
  "tag name" string
}



Table "images metadata" {
  id integer [primary key]
  "image id" integer [primary key]
  "image name" string 
}

Table "images blob store" {
  "image id"  integer [primary key]
  image blob
}



ref: accounts."user id" - carts."cart id"
ref: accounts."user id" < "orders metadata"."user id"
ref: products."product id" < "product tag mapping"."product id"
ref: "images metadata".id  > products."product id"
ref: "images metadata".id > "product tags"."tag id"
ref: "product tags"."tag id" < "product tag mapping"."tag id"
ref: carts."product id" > products."product id"
ref: products."product id" < "orders"."product id"
ref : orders."order id" > "orders metadata"."order id"
ref: "images blob store"."image id" - "images metadata"."image id"
