Project online_store {
  database_type: 'multi database'
  Note: 'Schema of each database'
}

Table accounts {
  "user id" integer [primary key]
  name string
  
    indexes {
    "user id" [pk]
  }
}

Table carts {
  "cart id" integer [primary key]
  "product id" integer [primary key]
  quantity integer
}

Table "orders metadata" {
  "user id" integer [primary key]
  "purchase date" date [primary key]
  "order id"  integer [primary key]
  price float
}

Table orders {
  "order id"  integer [primary key]
  "product id" integer [primary key]
  "quantity" integer
}

Table products {
  "product id" integer [primary key]
  title string
  description string
  
}

Table "product category mapping" {
  "product id" integer [primary key]
  "category id" integer [primary key]
  
}

Table categories {
  "category id" integer [primary key]
  "category name" string
}



Table images {
  "product or category id"  integer [primary key]
  purpose PurposeType [primary key]
  screen ScreenType [primary key]
  image blob 
}

Table prices {
  "price id" integer [primary key]
  price float
}



ref: accounts."user id" < carts."cart id"
ref: accounts."user id" < "orders metadata"."user id"
ref: products."product id" < "product category mapping"."product id"
ref: images."product or category id"  > products."product id"
ref: images."product or category id" > categories."category id"
ref: categories."category id" < "product category mapping"."category id"
ref: carts."product id" > products."product id"
ref: products."product id" < "orders"."product id"
ref : orders."order id" > "orders metadata"."order id"
ref: prices."price id" - products."product id"
